\NewDocumentCommand{\parseannotation}{m}
{
  \clist_map_inline:nn {#1}
  {
    \str_if_eq:nnT {##1} {thesisauthor} { \itemannotation{thesisauthor} }
    \str_if_eq:nnT {##1} {corresponding} { \itemannotation{corresponding} }
  }
}
